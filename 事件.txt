//全选、全不选函数
function allselect(){
     if(selectAll.prop('checked')){
        selectAllLabel.hide();
        deselectAllLabel.show();
        langs.prop('checked',true);
     }
     else{
        selectAllLabel.show();
        deselectAllLabel.hide();
        langs.prop('checked',false);
       }
}
//绑定全选、全不选函数
selectAll.change(allselect);
//反选
invertSelect.click(function(){
    langs.map(function(){
       $(this).prop('checked',!$(this).is(':checked'));
       }); 
    ifselect();
});  
//每个单选框都选中，全不选状态；取消一个，全选状态        
function ifselect(){
     if(langs.get().every(e=>e.checked)){
        selectAll.prop('checked',true);  allselect();
     }
    else{selectAll.prop('checked',false);
         selectAllLabel.show();
         deselectAllLabel.hide();} 
     }
 langs.click(ifselect); //给每个单选框绑定事件
